***Script top***
atividades: 
1 - levantar itensi a serem sincronizados
2 - levantar itens a serem excluídos
3 - envia resultado por e-mail
4 - insere na tabela dinamo total de dados copiados


$origem = "c:\dados"
$destino = "s3://empresa.cliente1"
$mailOrigem = "leandro@cloudtreinamentos.com"
$mailDestino = "le.porciuncula.rodrigues@gmail.com"

aws.exe s3 sync $origem $destino > C:\device-less\log.txt
date >> C:\device-less\log.txt
$mail = Get-Content -Path C:\device-less\log.txt
aws ses send-email --from $mailOrigem --to $mailDestino --subject "Backup Dados" --text "$mail"
